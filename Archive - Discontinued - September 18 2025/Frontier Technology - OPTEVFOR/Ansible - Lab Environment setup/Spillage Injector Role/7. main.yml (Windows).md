```bash
# tasks file for spillage injector role

- name: Ensure the user's Desktop directory exists
  win_file:
    path: C:\Users\{{ ansible_user }}\Desktop
    state: directory

- name: Remove existing Stargate credentials PDF if present
  win_file:
    path: C:\Users\{{ ansible_user }}\Desktop\stargate_credentials.pdf
    state: absent

- name: Copy Stargate credentials PDF to Desktop
  win_copy:
    src: /home/bugs/ansible_playbooks/files/stargate_credentials.pdf
    dest: C:\Users\{{ ansible_user }}\Desktop\stargate_credentials.pdf

- name: Ensure the file is not hidden
  win_shell: |
    attrib -h C:\Users\{{ ansible_user }}\Desktop\stargate_credentials.pdf

- name: Verify file exists on Desktop
  win_stat:
    path: C:\Users\{{ ansible_user }}\Desktop\stargate_credentials.pdf
  register: pdf_file

- name: Output message if PDF exists
  debug:
    msg: "Stargate credentials PDF has been successfully copied to the Desktop."
  when: pdf_file.stat.exists

- name: List files on Desktop
  win_shell: dir C:\Users\{{ ansible_user }}\Desktop
  register: dir_output

- name: Output the Desktop directory contents
  debug:
    var: dir_output.stdout_lines
```
