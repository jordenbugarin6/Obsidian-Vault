# The execution of this playbook creates a directory named `China Number One` and drops two files into it to show `cohabitation`  - the first is `redteamcredentials.txt` which can have some sort of credentials in it as well as `stargate_safe.txt` which is some attacker notes to show that there is another attacker on the system. The Linux version of this also creates the user ChinaHackerGodX.
---
- name: Linux Spillage Injector Role
  hosts: linux
  become: yes
  tasks:
    - name: Ensure user ChinaHackerGodX exists (Linux)
      ansible.builtin.user:
        name: ChinaHackerGodX
        state: present
        create_home: yes
        shell: /bin/bash
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Create directory /tmp/China_Number_One (Linux)
      ansible.builtin.file:
        path: /tmp/China_Number_One
        state: directory
        mode: '0755'
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Upload redteamcredentials.txt to /tmp/China_Number_One (Linux)
      ansible.builtin.copy:
        src: /home/bugs/ansible_playbooks/files/redteamcredentials.txt
        dest: /tmp/China_Number_One/redteamcredentials.txt
        mode: '0644'
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Upload stargate_safe.txt to /tmp/China_Number_One (Linux)
      ansible.builtin.copy:
        src: /home/bugs/ansible_playbooks/files/stargate_safe.txt
        dest: /tmp/China_Number_One/stargate_safe.txt
        mode: '0644'
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Verify redteamcredentials.txt is present (Linux)
      ansible.builtin.stat:
        path: /tmp/China_Number_One/redteamcredentials.txt
      register: creds_file
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Verify stargate_safe.txt is present (Linux)
      ansible.builtin.stat:
        path: /tmp/China_Number_One/stargate_safe.txt
      register: stargate_file
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Output message if both files exist (Linux)
      ansible.builtin.debug:
        msg: "Both redteamcredentials.txt and stargate_safe.txt have been deployed successfully on Linux"
      when: creds_file.stat.exists and stargate_file.stat.exists

    - name: List contents of /tmp/China_Number_One
      ansible.builtin.shell: |
        ls -l /tmp/China_Number_One
      register: dir_listing
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']

    - name: Output directory contents
      ansible.builtin.debug:
        var: dir_listing.stdout_lines
      when: ansible_facts['os_family'] in ['Debian', 'RedHat']
